              old_user = @advocate.as_json
              assert_no_difference('Advocate.count') do
                @advocate.destroy
              end

              %i[email firstname lastname siret ...].each do |field|
                assert_not_equal old_user[field], @advocate.send(field)
              end

              assert @advocate.deleted
            end
